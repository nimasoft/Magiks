92 def arc(traj = None, center_dw = 0.1, center_dh = 0.0, direction = None, angle = 360.0, d_theta = 10.0, clockwise = True, static_ends = True, in_degrees = True):
93  '''
94  adds a set of points to the given trajectory establishing an arc in a plane parallel to the yz plane starting from initial_pos,
95  The points follow an arc with given angle around the center
96  input trj must be a Trajectory_Polynomial
97  if trj is not given, a new trajectory will be created
98  '''
99  if in_degrees:
100  angle = angle*math.pi/180.0
101  d_theta = d_theta*math.pi/180.0
102 
103  (trj, w, h, n, v) = decode_arguments(traj, direction, static_ends)
104 
105  if clockwise:
106  sgn = 1
107  else:
108  sgn = - 1
109 
110  center = center_dw*w + center_dh*h
111 
112  nseg = len(trj.segment)
113  npnt = len(trj.segment[nseg-1].point)
114 
115  initial_pos = np.copy(trj.segment[nseg-1].point[npnt - 1].pos)
116  initial_phi = trj.phi_end
117 
118  r = math.sqrt(center_dw**2 + center_dh**2)
119  t0 = math.acos(- center_dw/r)
120  t = 0.0
121  stay = True
122  pos = np.zeros(3)
123  while stay:
124  t = t + d_theta
125  if t > angle:
126  t = angle
127 
128  pos = initial_pos + center + r*math.cos(sgn*t + t0)*w + r*math.sin(sgn*t + t0)*h
129  vel = - r*sgn*math.sin(sgn*t + t0)*w + r*sgn*math.cos(sgn*t + t0)*h
130  # trj.new_segment()
131  trj.add_point(phi = initial_phi + t, pos = np.copy(pos), vel = vel)
132 
133  if gen.equal(t, angle):
134  stay = False
135 
136  lsi = len(trj.segment) - 1
137  lpi = len(trj.segment[lsi].point) - 1
138  trj.segment[lsi].point[lpi].vel = v
139 
140  return trj
