248  def ik_direction(self, u = None):
249  '''
250  Returns the joint direction (joint update correction) expected to lead the endeffector closer to the target.
251  This direction should be multiplied by a proper stepsize to ensure that the pose error norm is reduced.
252  '''
253  genpy.check_valid(self.ik_settings.algorithm, all_algorithms, __name__, self.__class__.__name__, sys._getframe().f_code.co_name, 'ik_settings.algorithm')
254 
255  err = self.pose_error()
256  Je = self.error_jacobian()
257  W = self.joint_damping_weights()
258 
259  # right pseudo inverse of error jacobian is calculated and placed in Je_dagger
260  #(Je_dagger,not_singular) = mathpy.right_pseudo_inverse(Je)
261  if self.ik_settings.algorithm == "JI":
262  Je_dag = np.linalg.inv(Je)
263  elif self.ik_settings.algorithm == "JPI":
264  Je_dag = vecmat.weighted_pseudo_inverse(Je, W)
265  elif self.ik_settings.algorithm == "JT":
266  Je_dag = - Je.T
267  elif self.ik_settings.algorithm in ["DLS(CDF)", "DLS(ADF)"]:
268  # Je_dag = vecmat.right_dls_inverse(Je, self.damping_factor)
269  Je_dag = vecmat.weighted_dls_inverse(Je, W, k = self.damping_factor)
270  elif self.ik_settings.algorithm in ['DLS(MADF)']:
271  mu = self.manipulability()
272  if mu < self.ik_settings.manipulability_threshold:
273  self.damping_factor = self.ik_settings.initial_damping_factor*(1.0 - mu/self.ik_settings.manipulability_threshold)
274  Je_dag = vecmat.weighted_dls_inverse(Je, W, k = self.damping_factor)
275  else:
276  assert False
277 
278  # Joint Correction is calculated here:
279  delta_qs = - np.dot(Je_dag, err)
280  # print delta_qs
281 
282  if u != None:
283  assert len(u) == self.config_settings.DOF
284  J_dag_J = np.dot(Je_dag, Je)
285  In_minus_J_dag_J = np.eye(self.config_settings.DOF) - J_dag_J
286  dq = np.dot(In_minus_J_dag_J, u)
287  # print "Internal motion: ", dq
288  delta_qs += dq
289 
290  return delta_qs
