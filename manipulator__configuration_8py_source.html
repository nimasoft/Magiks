205  def update_joint_limit_jacobian_multipliers(self):
206  '''
207  joint_limit_multipliers are coefficients of forward and inverse mapping to/from limited to/from unlimited jointspace.
208  i.e: jmc_a is the derivative of q to q_star
209  (Refer also to the comments of method: initialize in this class)
210  This method calculates and updates the values of one of the joint limit multipliers: "jmc_c" which is called "joint_limit_jacobian_multipliers"
211  Each column of the error jacobian is multiplied by the corresponding element of vector "jmc_c": Jstar = Je * diag(jmc_c)
212  Since the coefficient "jmc_c" for a joint, depends on the current value of that joint, it should be updated everytime "q" changes
213  The other multipliers: "jmc_a", "jmc_b", "jmc_f" and "jmc_g" do not depend on the joints, therefore do not need to be updated
214  '''
215  func_name = "update_joint_limit_jacobian_multipliers"
216  for i in range(0,self.config_settings.DOF):
217  if self.config_settings.joint_handling[i] == 'NM':
218  self.jmc_c[i] = 1.0
219 
220  elif self.config_settings.joint_handling[i] == 'LM':
221  self.jmc_c[i] = 1.0 / self.jmc_a[i]
222 
223  elif self.config_settings.joint_handling[i] == 'TM':
224  self.jmc_c[i] = - math.sin(self.qstar[i])/self.jmc_a[i]
225 
226  elif self.config_settings.joint_handling[i] == 'TGM':
227  t = self.jmc_f[i] * math.cos(self.qstar[i]) + self.jmc_g[i]
228  self.jmc_c[i] = - 2.0 * math.sin(self.qstar[i])/(self.jmc_a[i]*(1 + t**2))
229  else:
230  assert False, genpy.err_str(__name__, self.__class__.__name__, 'update_joint_limit_jacobian_multipliers', self.config_settings.joint_handling[i] + " is not a valid value for joint_handling")

253  if not self.config_settings.joint_handling[i] == 'No Mapping':
254  assert self.joints_in_range() #check only for joint i

257  '''
258  The following code defines the joint limit multipliers
259  jmc_a and jmc_b are auxiliary coefficients which are used for conversion from "q" to "qstar" space
260  (Please refer to the joint limits section of the documentation associated with this code)
261 
262  To make sure that all joints are in their desired range identified as: (self.ql, self.qh), the real jointspace is mapped into an unlimited jointspace.
263 
264  "q " represent the real joint values which are in a limited jointpace
265  "qstar" represent the mapped joint values in unlimited jointspace
266 
267  The following code creates and defines "Joint Limit Multipliers". These are coefficients for this jointspace mapping - (jmc: Jointspace Mapping Coefficients)
268  in linear mapping:
269 
270  q = a * qstar + b
271  qs = (q - b) / a
272 
273  in trigonometric mapping:
274 
275  q = f * cos(qstar) + g
276  qstar = arccos [ (q - g) / f ]
277 
278  a, b, f and g are calculated in such a way that the real joint values which are in their feasible range are mapped into an equivalent qstar in an unlimited space (-pi, +pi)
279 
280  "jmc_c" is multiplied by the columns of the error jacobian matrix
281  '''
282  self.jmc_a = np.zeros((self.config_settings.DOF))
283  self.jmc_b = np.zeros((self.config_settings.DOF))
284  self.jmc_c = np.zeros((self.config_settings.DOF))
285  self.jmc_f = np.zeros((self.config_settings.DOF))
286  self.jmc_g = np.zeros((self.config_settings.DOF))
287 
288  ii = 0
289  for i in range(0,self.config_settings.njoint):
290  if self.config_settings.free[i]:
291  self.jmc_f[ii] = 0.5*(self.config_settings.qh[i] - self.config_settings.ql[i])
292  self.jmc_g[ii] = 0.5*(self.config_settings.qh[i] + self.config_settings.ql[i])
293 
294  if self.config_settings.joint_handling[ii] == 'NM':
295  self.jmc_a[ii] = 1.00000
296  self.jmc_b[ii] = 0.00000
297  elif self.config_settings.joint_handling[ii] == 'LM':
298  self.jmc_a[ii] = 2*math.pi/(self.config_settings.qh[i] - self.config_settings.ql[i])
299  self.jmc_b[ii] = math.pi * (self.config_settings.qh[i] + self.config_settings.ql[i])/(self.config_settings.ql[i] - self.config_settings.qh[i])
300  elif self.config_settings.joint_handling[ii] == 'TM':
301  self.jmc_a[ii] = 2/(self.config_settings.qh[i] - self.config_settings.ql[i])
302  self.jmc_b[ii] = (self.config_settings.qh[i] + self.config_settings.ql[i])/(self.config_settings.ql[i] - self.config_settings.qh[i])
303  elif self.config_settings.joint_handling[ii] == 'TGM':
304  th = math.tan(0.5*self.config_settings.qh[i])
305  tl = math.tan(0.5*self.config_settings.ql[i])
306  self.jmc_a[ii] = 2/(th - tl)
307  self.jmc_b[ii] = (tl + th)/(tl - th)
308  self.jmc_f[ii] = 0.5*(th - tl)
309  self.jmc_g[ii] = 0.5*(th + tl)
310  else:
311  assert False, genpy.err_str(__name__, self.__class__.__name__, 'initialize', self.config_settings.joint_handling[i] + " is not a valid value for joint_handling")

363 
364  '''
365  The correction of joints is always restricted by joint limits and maximum feasible joint speed
366  If you want to move the joints in a desired direction, how much are you allowed to move?
367  This function returns a feasible interval for the stepsize in the given direction.
368  The joint direction of change must be multiplied by a scalar value in this range so that the applied changes
369  are feasible.
370  '''
371  etta_l = []
372  etta_h = []
373 
374  for i in range(self.config_settings.DOF):
375  if not gen.equal(direction[i], 0.0):
376  if self.config_settings.limited[i]:
377  a = (self.config_settings.ql[i] - self.q[i])/direction[i]
378  b = (self.config_settings.qh[i] - self.q[i])/direction[i]
379  etta_l.append(gen.sign_choice(a, b, direction[i]))
380  etta_h.append(gen.sign_choice(b, a, direction[i]))
381 
382  a = delta_t*max_speed/direction[i]
383  etta_l.append(gen.sign_choice(-a, a, direction[i]))
384  etta_h.append(gen.sign_choice( a,-a, direction[i]))
385 
386  if (etta_l == []) or (etta_h == []):
387  return (0.0, 0.0)
