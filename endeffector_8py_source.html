198  Calculates and updates the actual positions and orientations of all task_points and task_frames according to the given configuration
199  '''
200  #Create pose vector:

202 
203  cnt = 0 #counter for pose error vector
204  k = 0 #counter for pose vector
205 
206  H = self.transfer_matrices()
207  # For task_points:
208  for tp in self.task_point:
209  # Calculating the position of each task point
210  tp.error.set_current(tp.position())
211  # Counting total number of constraints
212  cnt = cnt + len(tp.error.value())
213  #Arranging three position coordinates of each task_point in the endeffector pose vector
214  for j in range(0,3):
215  self.current_pose[k] = tp.ra[j]
216  k +=1

219  # For task_frame:

221  tf.error.set_current(tf.orientation())
222  #Counting total number of constraints
223  cnt = cnt + tf.error.settings.weight.shape[0]
224  #Arranging nine orientation coordinates of each reference_orientation (Elements of the Rotation Matrix) in the endeffector pose vector
225  ra = tf.orientation().matrix()

227  for j in range(0,3):
