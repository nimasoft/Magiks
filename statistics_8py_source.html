54 
55  Return a high-precision sum of the given numeric data. If optional
56  argument ``start`` is given, it is added to the total. If ``data`` is
57  empty, ``start`` (defaulting to 0) is returned.
58 
59 
60  Examples
61  --------
62 
63  >>> _sum([3, 2.25, 4.5, -0.5, 1.0], 0.75)
64  11.0
65 
66  Some sources of round-off error will be avoided:
67 
68  >>> _sum([1e50, 1, -1e50] * 1000) # Built-in sum returns zero.
69  1000.0
70 
71  Fractions and Decimals are also supported:
72 
73  >>> from fractions import Fraction as F
74  >>> _sum([F(2, 3), F(7, 5), F(1, 4), F(5, 6)])
75  Fraction(63, 20)
76 
77  >>> from decimal import Decimal as D
78  >>> data = [D("0.1375"), D("0.2108"), D("0.3061"), D("0.0419")]
79  >>> _sum(data)
80  Decimal('0.6963')
81 
82  Mixed types are currently treated as an error, except that int is
83  allowed.
84  """
85  # We fail as soon as we reach a value that is not an int or the type of
86  # the first value which is not an int. E.g. _sum([int, int, float, int])
87  # is okay, but sum([int, int, float, Fraction]) is not.
88  allowed_types = set([int, type(start)])
89  n, d = _exact_ratio(start)
90  partials = {d: n} # map {denominator: sum of numerators}
91  # Micro-optimizations.
92  exact_ratio = _exact_ratio
93  partials_get = partials.get
94  # Add numerators for each denominator.
95  for x in data:
96  _check_type(type(x), allowed_types)
97  n, d = exact_ratio(x)
98  partials[d] = partials_get(d, 0) + n
99  # Find the expected result type. If allowed_types has only one item, it
100  # will be int; if it has two, use the one which isn't int.
101  assert len(allowed_types) in (1, 2)
102  if len(allowed_types) == 1:
103  assert allowed_types.pop() is int
104  T = int
105  else:
106  T = (allowed_types - set([int])).pop()
107  if None in partials:
108  assert issubclass(T, (float, Decimal))
109  assert not math.isfinite(partials[None])
110  return T(partials[None])
111  total = Fraction()
112  for d, n in sorted(partials.items()):
113  total += Fraction(n, d)
114  if issubclass(T, int):
115  assert total.denominator == 1
116  return T(total.numerator)
117  if issubclass(T, Decimal):
118  return T(total.numerator)/total.denominator
119  return T(total)

262 def median_grouped(data, interval=1):
263  """"Return the 50th percentile (median) of grouped continuous data.
264 
265  >>> median_grouped([1, 2, 2, 3, 4, 4, 4, 4, 4, 5])
266  3.7
267  >>> median_grouped([52, 52, 53, 54])
268  52.5
269 
270  This calculates the median as the 50th percentile, and should be
271  used when your data is continuous and grouped. In the above example,
272  the values 1, 2, 3, etc. actually represent the midpoint of classes
273  0.5-1.5, 1.5-2.5, 2.5-3.5, etc. The middle value falls somewhere in
274  class 3.5-4.5, and interpolation is used to estimate it.
275 
276  Optional argument ``interval`` represents the class interval, and
277  defaults to 1. Changing the class interval naturally will change the
278  interpolated 50th percentile value:
279 
280  >>> median_grouped([1, 3, 3, 5, 7], interval=1)
281  3.25
282  >>> median_grouped([1, 3, 3, 5, 7], interval=2)
283  3.5
284 
285  This function does not check whether the data points are at least
286  ``interval`` apart.
287  """
288  data = sorted(data)
289  n = len(data)
290  if n == 0:
291  raise StatisticsError("no median for empty data")
292  elif n == 1:
293  return data[0]
294  # Find the value at the midpoint. Remember this corresponds to the
295  # centre of the class interval.
296  x = data[n//2]
297  for obj in (x, interval):
298  if isinstance(obj, (str, bytes)):
299  raise TypeError('expected number but got %r' % obj)
300  try:
301  L = x - interval/2 # The lower limit of the median interval.
302  except TypeError:
303  # Mixed type. For now we just coerce to float.
304  L = float(x) - float(interval)/2
305  cf = data.index(x) # Number of values below the median interval.
306  # FIXME The following line could be more efficient for big lists.
307  f = data.count(x) # Number of data points in the median interval.
308  return L + interval*(n/2 - cf)/f

353 def _ss(data, c=None):
354  """Return sum of square deviations of sequence data.
355 
356  If ``c`` is None, the mean is calculated in one pass, and the deviations
357  from the mean are calculated in a second pass. Otherwise, deviations are
358  calculated from ``c`` as given. Use the second case with care, as it can
359  lead to garbage results.
360  """
361  if c is None:
362  c = mean(data)
363  ss = _sum((x-c)**2 for x in data)
364  # The following sum should mathematically equal zero, but due to rounding
365  # error may not.
366  ss -= _sum((x-c) for x in data)**2/len(data)
367  assert not ss < 0, 'negative sum of square deviations: %f' % ss
368  return ss

373

375  values. The optional argument xbar, if given, should be the mean of
376  the data. If it is missing or None, the mean is automatically calculated.
377 
378  Use this function when your data is a sample from a population. To
379  calculate the variance from the entire population, see ``pvariance``.
380

382

384  >>> variance(data)

386

388  the optional second argument ``xbar`` to avoid recalculating it:
389

395  ``data``. Giving arbitrary values for ``xbar`` may lead to invalid or
396  impossible results.
