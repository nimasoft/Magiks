212  '''
213  Solves the trigonometric equation for t denoted by:
214  code = 1: parameters should be: [p1, p2, p3]
215  p1*cos(t) + p2*sin(t) + p3 = 0
216  code = 2:
217  p1*cos^2(t) + p2*sin^2(t) + p3*sin(t)*cos(t) + p4*sin(t) + p5*cos(t) + p6 = 0
218 
219  It returns the solution set as an array
220  If no solution exists, it returns an empty array
221  '''
222  solution_set = []
223  if code == 1:
224  R = math.sqrt(parameters[0]**2 + parameters[1]**2)
225  assert (not gen.equal(R, 0))
226  s = - parameters[2] / R
227  if in_domain_sin(s):
228  sai0 = math.atan2(parameters[0], parameters[1])
229  if gen.equal(math.sin(sai0), parameters[0]/R):
230  sai = sai0
231  elif equal(math.sin(sai0), - parameters[0]/R):
232  sai = sai0 + math.pi
233  else:
234  # If you came here, something is wrong !

237  t1 = arcsin(s) - sai

243  if gen.equal(parameters[0]*math.cos(t2) + parameters[1]*math.sin(t2) + parameters[2] , 0):

249  '''
250  solves a trigonometric system of equations
251 
252  x = cos(t0)*(A*sin(t1) + B*cos(t1)) (1)
253  y = sin(t0)*(A*sin(t1) + B*cos(t1)) (2)
254  z = -B*sin(t1) + A*cos(t1) (3)
255 
256  for two angles t0 and t1 and returns a set of paired solutions as tulips (t0,t1)
257 
258  The system is over constrained so a solution may not exist. In thic case, an empty set is returned.
259  '''
260  solution_set = []
261  t0 = []
262  t0.append(math.atan2(y, x))
263  t0.append(t0[0] + math.pi)
264 
265  t1 = solve_equation(1, [A, - B, - z])
266 
267  for i in range(0,len(t0)):
268  for j in range(0,len(t1)):
269  q = A*math.sin(t1[j]) + B*math.cos(t1[j])
270  p = A*math.cos(t1[j]) - B*math.sin(t1[j])
271  c0 = math.cos(t0[i])
272  s0 = math.sin(t0[i])
273  if (gen.equal(x , c0*q) and
274  gen.equal(y , s0*q) and
275  gen.equal(z , p)):
276  solution_set.append((t0[i],t1[j]))
