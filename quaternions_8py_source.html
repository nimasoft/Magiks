150 def unit_quaternion(TRM):
151  '''
152  Returns a vector (4 X 1) containing elements of the unit quaternion corresponding to transformation or rotation matrix TRM.
153  (TRM can be the 4*4 transformation matrix or 3*3 rotation matrix)
154  the first element of the output vector contains the scalar part of the unit quaternion and the last three elements represent the vectorial part
155 
156  '''
157 
158  uqn = numpy.zeros((4))
159  uqn[0] = 1.00
160  [u,v,w] = permutation_uvw(TRM)
161 
162  p = math.sqrt(1 + TRM[u,u] - TRM[v,v] - TRM[w,w])
163 
164  if (p > 0.000001):
165  uqn[u+1] = p/2.0
166  uqn[v+1] = (TRM[v,u] + TRM[u,v])/(2*p)
167  uqn[w+1] = (TRM[w,u] + TRM[u,w])/(2*p)
168  uqn[0] = (TRM[w,v] - TRM[v,w])/(2*p)
169 
170  if uqn[0] < 0:
171  uqn = - uqn
172 
173  assert abs(numpy.linalg.norm(uqn) - 1) < 0.000001
174  return uqn

179  Matrix TRMD is the time derivative of the given transfer or rotation matrix TRM
